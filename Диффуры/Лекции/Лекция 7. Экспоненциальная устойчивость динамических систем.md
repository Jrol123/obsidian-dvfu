	$\begin{cases} {x_1'=x_2} \\ {x_2'=-x_1} \end{cases}$

	$x_1^2+x_2^2=C$

	$\exists$ функция Ляпунова
	$(f,\nabla V) \le -\alpha V\quad \alpha > 0$
	$C_1 |x|^2\le V(x)\le C_2|x|^2$
