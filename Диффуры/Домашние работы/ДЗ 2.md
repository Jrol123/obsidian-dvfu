# №796~
## Условие
$$
\begin{cases}
x' = x - y + z \\
y' = x + y - z \\
z' = 2x - y + 0z
\end{cases}
$$
## Решение
$$
\begin{pmatrix}
1 & -1 & 1 \\
1 & 1 & -1 \\
2 & -1 & 0
\end{pmatrix}
$$
$$
\begin{vmatrix}
1 - \lambda & -1 & 1 \\
1 & 1 - \lambda & -1 \\
2 & -1 & -\lambda
\end{vmatrix} = (1 - \lambda)\lambda^{2} - (1 - \lambda)\lambda + 2\lambda - 2 = 0
$$
$$
\begin{align}
(1 - \lambda)\left[ \lambda^{2} - \lambda \right] -(2 - 2\lambda) = 0 \\
(1 - \lambda)\left[ \lambda^{2} - \lambda - 2 \right] = 0 \\ 
\end{align}
$$
$$
\begin{align}
\lambda_{1} = 1 \\
\lambda_{2} = 2 \\
\lambda_{3} = -1
\end{align}
$$

\end{pmatrix} = 0
$$

\end{cases}
$$
