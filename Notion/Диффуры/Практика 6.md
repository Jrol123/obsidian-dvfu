(y')^2 - y^2 = 0  
}$

$\mathsf{  
y' = \pm y  
}$

$\mathsf{  
\dfrac{dy}{dx} = y  
}$

\dfrac{dy} y = dx  
}$

$\mathsf{  
\ln(y) = \pm x + C  
}$

$\mathsf{  
y = Ce^{\pm x}  
}$

$\mathsf{  
\begin{cases}  
\mathsf{(y')^2 - y^2 = 0} \\

}$

$\mathsf{

# №2

## Условие

$\mathsf{  
8(y')^3 = 27y  
}$

$\mathsf{  
y' = \sqrt[3]{\dfrac{27} 8 y}  
}$

$\mathsf{  
y' = \dfrac 3 2 y^{\frac 1 3}  
}$

$\mathsf{  
\dfrac{dy}{\sqrt[3] y} = \dfrac 3 2 dx  
}$

$\mathsf{  
\sqrt[3]{y^2} = x + C  
}$

$\mathsf{  
y = \pm \sqrt{(x + C)^3}  
}$

  

$\mathsf{  
\begin{cases}  
\mathsf{8(y')^3 - 27y = 0} \\  
\mathsf{24(y')^2 = 0}  
\end{cases}  
}$

$\mathsf{  
y_0 = 0  
}$ - особое решение

$\mathsf{  
y(x_0) = \pm \sqrt{(x + C)^3} = y_0(x_0) = 0  
}$

$\mathsf{  
y'(x_0) = \pm \sqrt[\frac 3 2]{(x + C)} = y'_0(x_0) = 0  
}$
